97       public void do_the_display(String[] search_inputs) 
98       { 
99           //This method checks the hashmap allergy_list, and potentially adds items to the search_inputs array 
100          search_inputs = checkListedAllergens(search_inputs); 
101   
102          ArrayList<MenuItemDisplay> results = new ArrayList<>(); 
103   
104          //Do this search feature for every item in the search string array 
105          int id = 0; 
106          for(String search_input:search_inputs) 
107          { 
108              for (MenuItem m : master_list) 
109              { 
110                  if (m.get_menuitemingredients().contains(search_input)) 
111                  { 
112                      //If the menu item is not in the result list, add it 
113                      //Otherwise, simply add the allergen info into the display object 
114                      if(checkForDuplicates(results, m)) 
115                      { 
116                          results.add(new MenuItemDisplay(m)); 
117                          results.get(results.size() - 1).setAllergen(search_input.trim()); 
118                      } 
119                      else 
120                      { 
121                          results.get(findMenuItem(results, m)).setAllergen(search_input.trim()); 
122                      } 
123                  } 
124              } 
125              id++; 
126          } 
127   
128          search_results_display_list = new MenuItemDisplay[results.size()]; 
129          results.toArray(search_results_display_list); 
130   
131          ListAdapter lAdapter = new CustomAdapter(this, search_results_display_list); 
132          ListView lView = (ListView) findViewById(R.id.viewList); 
133          lView.setAdapter(lAdapter); 
134          hideKeyboard(this);

149      private String[] checkListedAllergens(String[] search_inputs) 
150      { 
151          ArrayList<String> temp = new ArrayList<>(); 
152          for(String s : search_inputs) 
153          { 
154              //If not already in the temp list, add it 
155              //Add code to check for duplicate 
156              if(checkForDuplicates(temp, s)) 
157              { 
158                  temp.add(s); 
159              } 
160              if(allergy_list.containsKey(s)) 
161              { 
162                  for(String t : allergy_list.get(s)) 
163                  { 
164                      if(checkForDuplicates(temp, t)) 
165                      { 
166                          temp.add(t); 
167                      } 
168                  } 
169              } 
170          } 
171          String[] return_array = new String[temp.size()]; 
172          return temp.toArray(return_array); 
173      }

189      private boolean checkForDuplicates(ArrayList<String> list, String item) 
190      { 
191          for(String s : list) 
192          { 
193              if(s.equals(item)) 
194              { 
195                  return false; 
196              } 
197          } 
198          return true; 
199      }

214      public static void hideKeyboard(Activity activity) { 
215          InputMethodManager imm = (InputMethodManager) activity.getSystemService(Activity.INPUT_METHOD_SERVICE); 
216          //Find the currently focused view, so we can grab the correct window token from it. 
217          View view = activity.getCurrentFocus(); 
218          //If no view currently has focus, create a new one, just so we can grab a window token from it 
219          if (view == null) { 
220              view = new View(activity); 
221          } 
222          imm.hideSoftInputFromWindow(view.getWindowToken(), 0); 
223      }
