3    import android.os.Bundle;

5    import android.os.Parcelable; 
6     
7    import java.io.Serializable; 
8    import java.util.HashMap;

10   public class MenuItem implements Parcelable { 
11    
12       private int _id; 
13       private String _menuitemname; 
14       private String _menuitemingredients; 
15       //This is where individual components of the dish are stored 
16       //i.e. whipped cream for a thunder, so if someone had an ultrasperse allergy, 
17       //you'd know to just leave out the whipped cream. 
18       private HashMap<String, String> _complexingredients; 
19    
20       public MenuItem() 
21       { 
22           _complexingredients = new HashMap<>(); 
23       } 
24    
25       public MenuItem(Parcel in) 
26       { 
27           _id = in.readInt(); 
28           _menuitemname = in.readString(); 
29           _menuitemingredients = in.readString(); 
30           Bundle b = in.readBundle(); 
31           _complexingredients = (HashMap<String, String>)b.getSerializable("map"); 
32       } 
33    
34       public MenuItem(String itemname, String itemingredients) 
35       { 
36           this._menuitemname = itemname; 
37           this._menuitemingredients = itemingredients; 
38           _complexingredients = new HashMap<>(); 
39       } 
40    
41       public int get_id() { 
42           return _id; 
43       } 
44    
45       public void set_id(int _id) { 
46           this._id = _id; 
47       } 
48    
49       public String get_menuitemname() { 
50           return _menuitemname; 
51       } 
52    
53       public void set_menuitemname(String _menuitemname) { 
54           this._menuitemname = _menuitemname; 
55       } 
56    
57       public String get_menuitemingredients() { 
58           return _menuitemingredients; 
59       } 
60    
61       public void set_menuitemingredients(String _menuitemingredients) { 
62           this._menuitemingredients = _menuitemingredients; 
63       } 
64    
65       public boolean checkIngredients(String search) 
66       { 
67           return _menuitemingredients.contains(search); 
68       } 
69    
70       public HashMap<String, String> get_complexingredients() { 
71           return _complexingredients; 
72       } 
73    
74       public String getFromComplexIngredients(String key) 
75       { 
76           return _complexingredients.get(key); 
77       } 
78    
79       public void set_complexingredients(String key, String value) 
80       { 
81           _complexingredients.put(key, value); 
82       } 
83       public void set_complexingredients(HashMap<String, String> _complexingredients) { 
84           this._complexingredients = _complexingredients; 
85       } 
86    
87       @Override 
88       public int describeContents() { 
89           return 0; 
90       } 
91    
92       @Override 
93       public void writeToParcel(Parcel dest, int flags) { 
94           dest.writeInt(_id); 
95           dest.writeString(_menuitemname); 
96           dest.writeString(_menuitemingredients); 
97           Bundle b = new Bundle(); 
98           b.putSerializable("map", _complexingredients); 
99           dest.writeBundle(b); 
100

102      {

107
