46       public MenuItem[] loadMenu(InputStream file) throws XmlPullParserException, IOException 
47       { 
48           parseFactory = XmlPullParserFactory.newInstance(); 
49           parser = parseFactory.newPullParser(); 
50          // in = assMan.open("menu.xml"); 
51           in = file; 
52           parser.setFeature(XmlPullParser.FEATURE_PROCESS_NAMESPACES, false); 
53           parser.setInput(in, null); 
54    
55           int eventType = parser.getEventType(); 
56           ArrayList<menu> menus = new ArrayList<>(); 
57           ArrayList<complexitem> complexities = new ArrayList<>(); 
58           ArrayList<sauce> sauces = new ArrayList<>(); 
59    
60           menu curMenu = null; 
61           complexitem curComplex = null; 
62           sauce curSauce = null; 
63           int whichClass = 0; 
64           while(eventType != XmlPullParser.END_DOCUMENT) 
65           { 
66               String name = null; 
67               switch(eventType) 
68               { 
69                   case XmlPullParser.START_TAG: 
70                       name = parser.getName(); 
71                       if("menuitem".equals(name)) 
72                       { 
73                           curMenu = new menu(); 
74                           whichClass = 1; 
75                       } 
76                       else if("complexitem".equals(name)) 
77                       { 
78                           curComplex = new complexitem(); 
79                           whichClass = 2; 
80                       } 
81                       else if("sauce".equals(name)) 
82                       { 
83                           curSauce = new sauce(); 
84                           whichClass = 3; 
85                       } 
86                       else if("name".equals(name)) 
87                       { 
88                           switch(whichClass) 
89                           { 
90                               case 1: 
91                                   curMenu.setName(parser.nextText()); 
92                                   break; 
93                               case 2: 
94                                   curComplex.setName(parser.nextText()); 
95                                   break; 
96                               case 3: 
97                                   curSauce.setName(parser.nextText()); 
98                                   break; 
99                           } 
100                      } 
101                      else if("basicingredients".equals(name)) 
102                      { 
103                          switch(whichClass) 
104                          { 
105                              case 1: 
106                                  curMenu.setBasicIngredients(parser.nextText()); 
107                                  break; 
108                              case 2: 
109                                  curComplex.setBasicIngredients(parser.nextText()); 
110                                  break; 
111                              case 3: 
112                                  curSauce.setBasicIngredients(parser.nextText()); 
113                                  break; 
114                          } 
115                      } 
116                      else if("complexingredients".equals(name)) 
117                      { 
118                          switch(whichClass) 
119                          { 
120                              case 1: 
121                                  curMenu.setComplexIngredients(parser.nextText()); 
122                                  break; 
123                              case 2: 
124                                  curComplex.setComplexIngredients(parser.nextText()); 
125                                  break; 
126                              case 3: 
127                                  curSauce.setComplexIngredients(parser.nextText()); 
128                                  break; 
129                          } 
130                      } 
131                      else if("sauces".equals(name)) 
132                      { 
133                          curMenu.setSauce(parser.nextText()); 
134                      } 
135                      break; 
136                  case XmlPullParser.END_TAG: 
137                      name = parser.getName(); 
138                      if("menuitem".equals(name)) 
139                      { 
140                          menus.add(curMenu); 
141                      } 
142                      else if("complexitem".equals(name)) 
143                      { 
144                          complexities.add(curComplex); 
145                      } 
146                      else if("sauce".equals(name)) 
147                      { 
148                          sauces.add(curSauce); 
149                      } 
150                      else if("name".equals(name) || "basicingredients".equals(name) || "complexingredients".equals(name) || "sauces".equals(name)) 
151                      { 
152                          //parser.nextTag(); 
153                      } 
154                      break; 
155                  default: 
156                      break; 
157              } 
158              eventType = parser.next(); 
159          } 
160          return recursiveParser(menus, complexities, sauces); 
161      }

163      public MenuItem[] recursiveParser(ArrayList<menu> m, ArrayList<complexitem> c, ArrayList<sauce> s) 
164      { 
165   
166          HashMap<String, complexitem> complexMap = new HashMap<>(); 
167          for(complexitem x : c) 
168          { 
169              complexMap.put(x.name, x); 
170          } 
171          HashMap<String, sauce> sauceMap = new HashMap<>(); 
172          for(sauce x : s) 
173          { 
174              sauceMap.put(x.name, x); 
175          } 
176          MenuItem[] final_menu = new MenuItem[m.size()]; 
177          int id = 0; 
178   
179          for(menu x : m) 
180          { 
181              final_menu[id] = new MenuItem(); 
182              final_menu[id].set_menuitemname(x.getName()); 
183              final_menu[id].set_menuitemingredients(x.getBasicIngredients()); 
184              if(x.complexIngredients != null && x.complexIngredients.length() > 0) 
185              { 
186                  String[] stuff = x.complexIngredients.split(","); 
187                  String temp = ""; 
188                  for(String thing : stuff) 
189                  { 
190                      temp = recurseComplexIngredients(thing, complexMap); 
191                      final_menu[id].set_complexingredients(thing, temp); 
192                      final_menu[id].set_menuitemingredients(final_menu[id].get_menuitemingredients()+temp); 
193                  } 
194              } 
195              id++; 
196          } 
197   
198          return final_menu; 
199      }

201      public String recurseComplexIngredients(String pointer, HashMap<String, complexitem> map) 
202      { 
203          String complex = map.get(pointer).getBasicIngredients(); 
204          if(map.get(pointer).hasComplex()) 
205          { 
206              String[] stuff = map.get(pointer).complexIngredients.split(","); 
207              for(String thing : stuff) 
208              { 
209                  complex = complex + recurseComplexIngredients(thing, map); 
210              } 
211          } 
212          return "," + complex; 
213      } 
214   
215      public String recurseSauces(String pointer, HashMap<String, sauce> map, HashMap<String, complexitem> cmap) 
216      { 
217          String complex = ""; 
218          if(map.get(pointer).hasComplex()) 
219          { 
220              String[] stuff = map.get(pointer).complexIngredients.split(","); 
221              for(String thing : stuff) 
222              { 
223                  complex = complex + recurseSauces(thing, map, cmap); 
224              } 
225          } 
226          return "," + complex; 
227      }
